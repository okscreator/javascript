Задание №2.1

Почему код даёт именно такие результаты?

var a = 1, b = 1, c, d;
c = ++a; alert(c); //1 + 1 = 2 Префиксная форма инкремента возвращает результат сложения.
d = b++; alert(d); //1 Постфиксная форма инкремента возвращает значение до увеличения, но сохраняет после себя сумму b=b+1=1+1=2.
c = (2 + ++a); alert(c); //2 + 2 + 1 = 5 Префиксная форма инкремента с данными a записанными при вычислении с.
d = (2 + b++); alert(d); //2 + 2 = 4 Постфиксная форма инкремента с данными b записанными после вычислении d, но сохраняет после себя сумму b=b+1=2+1=3
alert(a); // 3 Результат а из строки с вычисления c.
alert(b); // 3 Результат b из строки с вычисления d.
